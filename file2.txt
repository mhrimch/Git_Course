
import { useUuiService, UuiServiceClientDef, UuiServiceResponse } from '@finanzinformatik/cpl-module-reactlib';
import {
  ExtendedUuiServiceMessage,
  NotificationStoreActions,
  useNotificationStoreActions,
} from '@personalsysteme/notification-handler';
import { queryOptions, useQuery, UseQueryResult } from '@tanstack/react-query';
import { KalenderRead } from '../../apl/pzw/kalenderUiModule/KalenderRead';
import { callReadKalender } from '../../apl/pzw/kalenderUiModule/KalenderUiModule.api';
import { TanstackConstants } from '../../constants/TanstackConstants';
import isRolloutVerfuegbar from '../../utils/rolloutUtils';
import { KalenderKeys } from './Kalender.keys';

export function readKalenderOptions(
  uuiService: Readonly<UuiServiceClientDef>,
  notificationStoreActions: NotificationStoreActions,
  beginn: string,
  ende: string
) {
  return queryOptions({
    queryKey: KalenderKeys.zeitraum(beginn, ende),
    queryFn: async (): Promise<KalenderRead & { isRollout: boolean }> =>
      callReadKalender(uuiService, beginn, ende).then((response: UuiServiceResponse<KalenderRead>) => {
        notificationStoreActions.handleMessages(response.messages as ExtendedUuiServiceMessage[], true);
        //TODO Entfernen wenn Rollout abgeschlossen ist
        return {
          ...(response.payload as KalenderRead),
          isRollout: isRolloutVerfuegbar(response.messages),
        };
      }),
    staleTime: TanstackConstants.LONG_STALE_TIME,
    placeholderData: (prev) => prev,
  });
}

export function useReadKalender(
  beginn: string,
  ende: string
): UseQueryResult<(KalenderRead & { isRollout: boolean }) | undefined> {
  const uuiService = useUuiService(),
    notificationStoreActions = useNotificationStoreActions();
  return useQuery(readKalenderOptions(uuiService, notificationStoreActions, beginn, ende));
}

     * Die Funktion readKalender liest den Kalender für den angemeldeten Nutzer.

     * @param beginn LocalDate
     * @param ende   LocalDate
     * @return kalenderRead KalenderRead
     * @throws UiServiceException UiServiceException

    public KalenderRead readKalender(
            LocalDate beginn,
            LocalDate ende
    ) throws UiServiceException {
        try {
            LiesPmgtPzwKalenderUeberUserEingabe eingabe = new LiesPmgtPzwKalenderUeberUserEingabe();
            LocalDateTime localDateTimeAnfang = beginn.atTime(0, 0, 0);
            LocalDateTime localDateTimeEnde = ende.atTime(23, 59, 59);
            eingabe.setTsGueltigAb(new IdTimestamp(localDateTimeAnfang.format(DateTimeFormatter.ofPattern(DateFormatConstants.TIMESTAMP_FORMAT, Locale.GERMANY))));
            eingabe.setTsGueltigBis(new IdTimestamp(localDateTimeEnde.format(DateTimeFormatter.ofPattern(DateFormatConstants.TIMESTAMP_FORMAT, Locale.GERMANY))));
            eingabe.setPctlPersSchl(userService.getPersonalnummer());
            return pmgtPzwKalenderLesenRepository.liesPmgtPzwKalenderUeberUser(eingabe).stream().findFirst().map(KalenderMapper::ospeToKalenderRead).orElse(null);
        } catch (OspeException e) {
            throw new UiServiceException(MessageType.ERROR, MessageCode.ERROR,
                    "Kalender.Read.HttpStatus.502",
                    MessageMetadata.instance(MessageCallType.READ, MessageScope.APL));
        }
    }
}

public class KalenderMapper {
    /**
     * Die Funktion ospeToKalenderRead konvertiert ein List<PmgtPzwKalenderWoche> in ein KalenderRead
     *
     * @param pmgtPzwKalenderLesenAusgabe PmgtPzwKalenderLesenAusgabe
     * @return KalenderRead
     */
    public static KalenderRead ospeToKalenderRead(PmgtPzwKalenderLesenAusgabe pmgtPzwKalenderLesenAusgabe) {
        return new KalenderRead(pmgtPzwKalenderLesenAusgabe.getPmgtPzwKalenderMitarbeiter().getPctlPersSchl(),
                pmgtPzwKalenderLesenAusgabe.getPmgtPzwKalenderMitarbeiter().getPzwMaAfbrName(),
                pmgtPzwKalenderLesenAusgabe.getPmgtPzwKalenderMitarbeiter().getPmgtPzwKalenderWoche().stream().flatMap(kalenderwoche -> ospePmgtPzwKalenderWocheToKalenderTage(kalenderwoche).stream()).toList());
    }

    public static List<KalendertagRead> ospePmgtPzwKalenderWocheToKalenderTage(PmgtPzwKalenderWoche pmgtPzwKalenderWoche) {
        return pmgtPzwKalenderWoche.getPmgtPzwKalenderTag().stream().map(tag -> ospeToKalenderTag(tag, pmgtPzwKalenderWoche.getPzwKw())).toList();
    }

    public static KalendertagRead ospeToKalenderTag(PmgtPzwKalenderTag pmgtPzwKalenderTag, String pzwKw) {
        return new KalendertagRead(DateUtils.convertToLocalDate(pmgtPzwKalenderTag.getPzwKtag()),
                pzwKw,
                PskConstants.JA.equals(pmgtPzwKalenderTag.getPzwKtagFtagKz()),
                PzwConstants.ANWESENHEITSSTATUS_OHNE_MELDUNG.equals(pmgtPzwKalenderTag.getPzwAnwhStat()),
                pmgtPzwKalenderTag.getPzwSdoWert(),
                ospeToKalenderAnOderAbwesenheit(pmgtPzwKalenderTag),
                pmgtPzwKalenderTag.getPmgtPzwKalenderTagBuchung().stream().map(KalenderMapper::ospeToBuchungspaar).toList(),
                pmgtPzwKalenderTag.getPmgtPzwKalenderTagBuchungOffen().stream().map(KalenderMapper::ospeToBuchungRead).toList(),

                pmgtPzwKalenderTag.getPmgtPzwKalenderTagAntragOffen().stream().map(KalenderMapper::ospeToKalenderAntragOffenRead).toList(),
                pmgtPzwKalenderTag.getPmgtPzwKalenderTagMeldung().stream().map(KalenderMapper::ospeToMeldung).toList()

        );
    }
