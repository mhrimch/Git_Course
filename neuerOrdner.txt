  @AplLogging(logExecutionTime = true, includeParameters = true, logTrace = true)
    public List<AntragAbwesenheitRead> readAntraegeAbwesenheit(
            LocalDate beginn,
            LocalDate ende
    ) throws UiServiceException {
        try {
            LocalDateTime localDateTimeAnfang = beginn.atTime(0, 0, 0);
            LocalDateTime localDateTimeEnde = ende.atTime(23, 59, 59);
            DateTimeFormatter tsFormatter = DateTimeFormatter.ofPattern((DateFormatConstants.TIMESTAMP_FORMAT), Locale.GERMANY);

            LiesPmgtPzwAntragUeberUserEingabe eingabe = new LiesPmgtPzwAntragUeberUserEingabe();

            eingabe.setPctlPersSchl(userService.getPersonalnummer());
            eingabe.setTsGueltigAb(new IdTimestamp(localDateTimeAnfang.format(tsFormatter)));
            eingabe.setTsGueltigBis(new IdTimestamp(localDateTimeEnde.format(tsFormatter)));
            eingabe.setPzwAntrArt(PzwAntraegeConstants.ANTRAGSART_ABWESENHEITEN);

            Optional<PmgtPzwAntragLesenAusgabe> antragLesenAusgabe = antragLesenRepository
                    .liesPmgtPzwAntragUeberUser(eingabe);

            Optional<PmgtPzwAntragAbwht> antrag = antragLesenAusgabe.flatMap(x -> x.getPmgtPzwAntragMitarbeiter().getPmgtPzwAntragAbwht().stream().findFirst());

            IdTimestamp massVgngId = null;
            if (antrag.isPresent()) {
                massVgngId = antrag.get().getMassVgngId();
            }

            List<DokumentRead> dokumenteListe = new ArrayList<>();

            if (!Objects.equals(massVgngId, null) && !Objects.equals(massVgngId, new IdTimestamp(PzwAntraegeConstants.LOW_TIMESTAMP))) {
                PmgtPskDkmtDatenLesenAusgabe dokumente = dokumentService.readPmgtDokumentUeberMassVgngId(massVgngId);

                if (dokumente != null && dokumente.getPmgtPskDkmtDaten() != null) {
                    for (var dkmtDaten : dokumente.getPmgtPskDkmtDaten()) {
                        var detail = dokumentService.readPmgtDokument(dkmtDaten.getPmgtElekDkmtId());

                        if (detail != null) {
                            dokumenteListe.add(new DokumentRead(
                                    detail.dokumentName(),
                                    detail.undDownloadToken()
                            ));
                        }
                    }
                }
            }

            return antragLesenAusgabe
                    .map(ausgabe -> AntraegeMapper.ospeToantragAbwesenheitRead(ausgabe, dokumenteListe))
                    .orElse(null);

        } catch (OspeException e) {
            throw new UiServiceException(MessageType.ERROR, MessageCode.ERROR,
                    "Antraege.Read.HttpStatus.502",
                    MessageMetadata.instance(MessageCallType.READ, MessageScope.APL));
        }
    }

    public static List<AntragAbwesenheitRead> ospeToantragAbwesenheitRead(PmgtPzwAntragLesenAusgabe ausgabe, List<DokumentRead> dokumente) {
        return Optional.ofNullable(ausgabe.getPmgtPzwAntragMitarbeiter())
                .stream()
                .flatMap(antragMitarbeiter -> antragMitarbeiter.getPmgtPzwAntragAbwht().stream())
                .map(antrag -> new AntragAbwesenheitRead(
                        antrag.getPzwAntrZrmAts(),
                        antrag.getPzwAntrZrmEts(),
                        PskConstants.JA.equals(antrag.getPzwAntrStrnKz()),
                        antrag.getPzwAbwtSchl(),
                        antrag.getPzwAntrTageAnz().toBigInteger().intValue(),
                        Abwesenheitszyklus.GANZTAGS.getZyklusSchluessel().equals(antrag.getPzwAntrZyklSchl()),
                        antrag.getPzwAntrMaHnws(),
                        antrag.getPzwAntrFrgbHnws(),
                        antrag.getTsEinfuegung(),
                        AntragAbwesenheitStatus.getAntragStatus(antrag.getPzwAntrStat()),
                        antrag.getAbsVgngPrzId(),
                        antrag.getMassVgngId(),
                        dokumente
                )).toList();
    }
