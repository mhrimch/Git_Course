dayjs.extend(weekOfYear);
export type AnsichtArt = 'Arbeitswoche' | 'Woche' | 'Monat';

export default function Kalender() {
  const [kalenderansicht, setKalenderansicht] = useState<AnsichtArt>('Arbeitswoche'),
    [beginn, setBeginn] = useQueryState(
      KalenderConstants.QUERY_KEYS.BEGINN,
      parseAsString
        .withDefault(dayjs().day(KalenderConstants.MONTAG).format(KalenderConstants.DATUM_ISO_FORMAT_STRING))
        .withOptions({ clearOnDefault: false })
    ),
    [ende, setEnde] = useQueryState(
      KalenderConstants.QUERY_KEYS.ENDE,
      parseAsString
        .withDefault(dayjs().day(KalenderConstants.FREITAG).format(KalenderConstants.DATUM_ISO_FORMAT_STRING))
        .withOptions({ clearOnDefault: false })
    ),
    [selectedTag, setSelectedTag] = useState<string | undefined>(
      dayjs().format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
    ),
    [selectedMonatBeginn, setSelectedMonatBeginn] = useState<string>(
      dayjs().startOf(KalenderConstants.MONAT).format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
    ),
    { data: kalenderData, isFetching: isKalenderFetching } = useReadKalender(beginn, ende),
    grenzeMonateZurueck = dayjs()
      .startOf(KalenderConstants.MONAT)
      .subtract(KalenderConstants.MONATS_GRENZE_ZURUECK, KalenderConstants.MONAT),
    grenzeMonateVor = dayjs()
      .startOf(KalenderConstants.MONAT)
      .add(KalenderConstants.MONATS_GRENZE_VOR, KalenderConstants.MONAT),
    { t } = useTranslation(),
    workflowService = useWorkflowService();
  useEffect(() => {
    if (kalenderansicht === KalenderConstants.ANSICHTARTEN.MONAT) {
      updateGrenzenMonatsansicht();
    }
  }, [selectedMonatBeginn, kalenderansicht]);
  const handleChangeKalenderansicht = (ansichtart: AnsichtArt) => {
    setKalenderansicht(ansichtart);
    resetBeginnUndEnde(ansichtart);
  };
  const resetBeginnUndEnde = (ansichtart: AnsichtArt) => {
    if (ansichtart === KalenderConstants.ANSICHTARTEN.MONAT) {
      setSelectedMonatBeginn(
        dayjs().startOf(KalenderConstants.MONAT).format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
      );
    } else if (ansichtart === KalenderConstants.ANSICHTARTEN.WOCHE) {
      setBeginn(dayjs().day(KalenderConstants.MONTAG).format(KalenderConstants.DATUM_ISO_FORMAT_STRING));
      setEnde(
        dayjs().day() === KalenderConstants.SONNTAG
          ? dayjs().format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
          : dayjs().day(KalenderConstants.SONNTAG_FOLGEWOCHE).format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
      );
    } else {
      setBeginn(dayjs().day(KalenderConstants.MONTAG).format(KalenderConstants.DATUM_ISO_FORMAT_STRING));
      setEnde(dayjs().day(KalenderConstants.FREITAG).format(KalenderConstants.DATUM_ISO_FORMAT_STRING));
    }
  };

  const addToDatum = (addValue: number) => {
    setBeginn(dayjs(beginn).add(addValue, KalenderConstants.WOCHE).format(KalenderConstants.DATUM_ISO_FORMAT_STRING));
    setEnde(dayjs(ende).add(addValue, KalenderConstants.WOCHE).format(KalenderConstants.DATUM_ISO_FORMAT_STRING));
  };
  const subtractFromDatum = (subtractValue: number) => {
    setBeginn(
      dayjs(beginn).subtract(subtractValue, KalenderConstants.WOCHE).format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
    );
    setEnde(
      dayjs(ende).subtract(subtractValue, KalenderConstants.WOCHE).format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
    );
  };
  const updateGrenzenMonatsansicht = () => {
    setBeginn(
      dayjs(selectedMonatBeginn).startOf(KalenderConstants.MONAT).day() === KalenderConstants.SONNTAG
        ? dayjs(selectedMonatBeginn)
            .day(KalenderConstants.MONTAG_VORWOCHE)
            .format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
        : dayjs(selectedMonatBeginn).day(KalenderConstants.MONTAG).format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
    );
    setEnde(
      dayjs(selectedMonatBeginn).endOf(KalenderConstants.MONAT).day() === KalenderConstants.SONNTAG
        ? dayjs(selectedMonatBeginn).endOf(KalenderConstants.MONAT).format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
        : dayjs(selectedMonatBeginn)
            .endOf(KalenderConstants.MONAT)
            .day(KalenderConstants.SONNTAG_FOLGEWOCHE)
            .format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
    );
  };
  const handleClickVorwaerts = () => {
    if (kalenderansicht === KalenderConstants.ANSICHTARTEN.MONAT) {
      setSelectedMonatBeginn((prevState) =>
        dayjs(prevState).add(1, KalenderConstants.MONAT).format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
      );
    } else {
      addToDatum(KalenderConstants.EINHEIT_WOCHE);
    }
    setSelectedTag(undefined);
  };
  const handleClickZurueck = () => {
    if (kalenderansicht === KalenderConstants.ANSICHTARTEN.MONAT) {
      setSelectedMonatBeginn((prevState) =>
        dayjs(prevState).subtract(1, KalenderConstants.MONAT).format(KalenderConstants.DATUM_ISO_FORMAT_STRING)
      );
    } else {
      subtractFromDatum(KalenderConstants.EINHEIT_WOCHE);
    }
    setSelectedTag(undefined);
  };
  const handleClickHeute = () => {
    setSelectedTag(dayjs().format(KalenderConstants.DATUM_ISO_FORMAT_STRING));
    resetBeginnUndEnde(kalenderansicht);
  };
  const selectedMonat = () => {
    return dayjs(selectedMonatBeginn).locale(PzwConstants.LOCALES_DE).format(KalenderConstants.MONAT_FORMAT_STRING);
  };
  const kalendertag = kalenderData?.kalendertage.find((tag) => tag.datum === selectedTag);
  if (!kalenderData?.isRollout) {
    return <></>;
  }
  const zurueckMoeglich = () => {
    if (kalenderansicht === KalenderConstants.ANSICHTARTEN.MONAT) {
      return dayjs(selectedMonatBeginn).subtract(1, KalenderConstants.MONAT).isSameOrAfter(grenzeMonateZurueck);
    }
    return dayjs(beginn)
      .subtract(KalenderConstants.EINHEIT_WOCHE, KalenderConstants.WOCHE)
      .isSameOrAfter(grenzeMonateZurueck);
  };
  const vorwaertsMoeglich = () => {
    if (kalenderansicht === KalenderConstants.ANSICHTARTEN.MONAT) {
      return dayjs(selectedMonatBeginn).add(1, KalenderConstants.MONAT).isSameOrBefore(grenzeMonateVor);
    }
    return dayjs(ende).add(KalenderConstants.EINHEIT_WOCHE, KalenderConstants.WOCHE).isSameOrBefore(grenzeMonateVor);
  };
  return (
    <Stack spacing={ThemeConstants.GRID_CONTAINER_SPACING}>
      <Box>
        <Button startIcon={<IcBackArrowS />} onClick={() => navigateToZeitwirtschaft(workflowService)}>
          {t('ZurueckZuZeitwirtschaft')}
        </Button>
      </Box>
      <KalenderToolbar
        kalenderansicht={kalenderansicht}
        handleChangeKalenderansicht={handleChangeKalenderansicht}
        handleClickZurueck={zurueckMoeglich() ? handleClickZurueck : undefined}
        handleClickVorwaerts={vorwaertsMoeglich() ? handleClickVorwaerts : undefined}
        selectedKwoderMonat={
          kalenderansicht === KalenderConstants.ANSICHTARTEN.MONAT
            ? selectedMonat()
            : t('KW') + ' ' + dayjs(beginn).week().toString()
        }
        handleClickHeute={handleClickHeute}
        anfangDatum={beginn}
        endeDatum={ende}
      />
      {isKalenderFetching ? (
        <KalenderGridSkeleton kalenderansicht={kalenderansicht}></KalenderGridSkeleton>
      ) : (
        <KalenderGrid
          kalenderData={kalenderData}
          kalenderansicht={kalenderansicht}
          selectedTag={selectedTag}
          setSelectedTag={setSelectedTag}
        ></KalenderGrid>
      )}
      {kalendertag && <KalenderDetails kalendertag={kalendertag} widget={false} />}
    </Stack>
  );
}

export type AntraegeRow = {
  einfuegungDatum: Dayjs;
  einfuegungTimestamp: IdTimestamp;

export function parseReadAntraegeAbwesenheitToAntraegeTabelleRow(
  data: AntragAbwesenheitRead[],
  schluesselPropertiesMap: SchluesselPropertiesMap
): AntraegeRow[] {
  const { DATUM_FORMAT_STRING, UHRZEIT_FORMAT_STRING } = KalenderConstants;
  return data
    .map(
      ({
        einfuegungDatum,
        antragsSchluessel,
        timestampVon,
        timestampBis,
        isGanztaegig,
        anzahlTage,
        antragStatus,
        mitarbeiterKommentar,
        fuehrungskraftKommentar,
        absVgngPrzId,
        isStorniert,
      }) => {
        const schluesselProperties = schluesselPropertiesMap[antragsSchluessel];
        const dayjsBeginn = idTimestampToDayjs(timestampVon);
        const dayjsEnde = idTimestampToDayjs(timestampBis);
        if (schluesselProperties !== undefined) {
          return {
            einfuegungDatum: idTimestampToDayjs(einfuegungDatum),
            einfuegungTimestamp: einfuegungDatum,
            typ: isStorniert
              ? schluesselProperties.antragBezeichnung + ' - ' + t('Stornierung')
              : schluesselProperties.antragBezeichnung,
            anfangTs: timestampVon,
            endeTs: timestampBis,
            zeitraum: isGanztaegig
              ? `${dayjsBeginn.format(DATUM_FORMAT_STRING)} - ${dayjsEnde.format(DATUM_FORMAT_STRING)}`
              : `${dayjsBeginn.format(DATUM_FORMAT_STRING)} ${dayjsBeginn.format(UHRZEIT_FORMAT_STRING)} - ${dayjsEnde.format(UHRZEIT_FORMAT_STRING)}`,
            status: antragStatus,
            antragSchluessel: antragsSchluessel,
            antragGruppeId: schluesselProperties.gruppeId,
            iconId: schluesselProperties.iconId,
            antragBezeichnung: schluesselProperties.antragBezeichnung,
            anzahlTage,
            mitarbeiterKommentar,
            fuehrungskraftKommentar,
            isGanztaegig,
            absVgngPrzId,
            isStornierung: isStorniert,
          };
        }
        return undefined;
      }
    )
    .filter((item) => item !== undefined) as AntraegeRow[];
}

type SchluesselProperties = {
  iconId: number;
  kategorie: string;
  gruppe: string;
  gruppeId: string;
  antragBezeichnung: string;
};
type SchluesselPropertiesMap = Record<string, SchluesselProperties | undefined>;
export function getSchluesselToPropertiesMap(konfAntraege: KonfAntragskategorieRead[]): SchluesselPropertiesMap {
  const mapper: SchluesselPropertiesMap = {};
  for (const kategorie of konfAntraege) {
    for (const gruppe of kategorie.antragsgruppen) {
      for (const schluessel of gruppe.antraege) {
        mapper[schluessel.schluessel] = {
          kategorie: kategorie.bezeichnung,
          gruppe: gruppe.bezeichnung,
          gruppeId: gruppe.id,
          iconId: gruppe.iconId,
          antragBezeichnung: schluessel.bezeichnung,
        };
      }
    }
  }
  return mapper;
}

export default function AntraegeAbwesenheitTabelle({ hidden, anfangDatum, endeDatum }: ZeitereignisseTabelleProps) {

          beginn: antrag.anfangTs,

  const endeIso = endeDayjs.format(KalenderConstants.DATUM_ISO_FORMAT_STRING);

  const { data: antraegeRead, isPending: isAntragPending } = useReadAntraegeAbwesenheit(anfangIso, endeIso);
  const konfAntraege = useReadKonfAntraege().data?.antragskategorien ?? [];
  const schluesselPropertiesMap = useMemo(() => getSchluesselToPropertiesMap(konfAntraege), [konfAntraege]);
  const parsedData = antraegeRead
    ? parseReadAntraegeAbwesenheitToAntraegeTabelleRow(antraegeRead, schluesselPropertiesMap)
    : [];
  const currentAntrag = erstellungszeitpunkt
    ? parsedData.find(
        (antrag) =>
          antrag.anfangTs === beginn && antrag.endeTs === ende && antrag.einfuegungTimestamp === erstellungszeitpunkt
      )
    : parsedData.find(
        (antrag) =>
          antrag.anfangTs === beginn &&
          antrag.endeTs === ende &&
          antrag.status === AntragAbwesenheitStatusConstants.FREIGEGEBEN_SAP
      );
  const isUrlaubsantrag = currentAntrag?.antragSchluessel === Antragsschluessel.urlaubsantrag;
  if (!isUrlaubsantrag && currentAntrag?.anzahlTage) {
    currentAntrag.anzahlTage = endeDayjs.diff(anfangDayjs, KalenderConstants.TAG);
  }
  if (!isAntragPending && !currentAntrag && inputModel.bezeichnung && inputModel.iconNr) {
    return (
      <KalenderViewGrid>
        {
          <AntragAbwesenheitSapAnzeigen
            antragBezeichnung={inputModel.bezeichnung}
            anfangTs={inputModel.beginn}
            endeTs={inputModel.ende}
            iconId={inputModel.iconNr}
          />
        }
      </KalenderViewGrid>
    );
  }
  return (
    <KalenderViewGrid>
      {currentAntrag && (
        <AntragAbwesenheitAnzeigen antragData={currentAntrag} dokumente={antraegeRead?.[0].dokumente} />
      )}
    </KalenderViewGrid>
  );
}
