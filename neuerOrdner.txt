type Tables = 'zeitkonto' | 'antraege' | 'auswertung';
export type ZeitereignisseTabelleProps = {
  hidden: boolean;
  anfangDatum: string;
  endeDatum: string;
};
export default function ZeitereignisseView() {
  const [table, setTable] = useState<Tables>('zeitkonto');
  const buttons: { id: Tables; title: string }[] = [
    { id: 'zeitkonto', title: t('zeitkonto') },
    { id: 'antraege', title: t('antraege') },
    { id: 'auswertung', title: t('auswertung') },
  ];
  const theme = useTheme();
  const workflowservice = useWorkflowService();
  const { anfangDatum, endeDatum } = useInputModel<ZeitereignissePageModel>() as ZeitereignissePageModel;
  let anfangDatumQuery = anfangDatum;
  let endeDatumQuery = endeDatum;
  if (!anfangDatum || !endeDatum) {
    const wocheAnfang = dayjs().day(KalenderConstants.MONTAG).format(KalenderConstants.DATUM_ISO_FORMAT_STRING);
    const wocheEnde = dayjs().day(KalenderConstants.FREITAG).format(KalenderConstants.DATUM_ISO_FORMAT_STRING);
    anfangDatumQuery = wocheAnfang;
    endeDatumQuery = wocheEnde;
  }
  return (
    <>
      <KalenderViewGrid>
        <Stack gap={10}>
          <Stack alignItems="center" direction="row" spacing={2}>
            <ButtonBase
              onClick={() => {
                navigateToZeitwirtschaft(workflowservice);
              }}
            >
              <Stack alignItems="center" direction="row" spacing={2}>
                <IcBackArrowS />
                <Typography variant="button01UnderlineLight">{t('zurueckZeitwirtschaft')}</Typography>
              </Stack>
            </ButtonBase>
          </Stack>

          <Stack direction={'row'} justifyContent={'space-between'} alignItems="flex-start">
            <Stack spacing={2}>
              <Typography variant={ThemeConstants.TYPOGRAPHY_HEADLINE02}>{t('zeitereignisseUndAntraege')}</Typography>
              <Typography variant={ThemeConstants.TYPOGRAPHY_HEADLINE04}>
                {anfangDatum} - {endeDatumQuery}
              </Typography>
            </Stack>
            <KalenderNavToggleButton value="Zeitereignisse" />
          </Stack>

          <Stack alignItems="center" direction="row" spacing={2}>
            {buttons.map((btn) => (
              <Button
                key={btn.id}
                onClick={() => setTable(btn.id)}
                variant="outlined"
                size="small"
                sx={
                  btn.id === table
                    ? {
                        color: 'white',
                        backgroundColor: theme.palette.secondary.main,
                      }
                    : {}
                }
              >
                {btn.title}
              </Button>
            ))}
          </Stack>
          <ZeitkontoTabelle hidden={table !== 'zeitkonto'} anfangDatum={anfangDatumQuery} endeDatum={endeDatumQuery} />
          <AntraegeAbwesenheitTabelle
            hidden={table !== 'antraege'}
            anfangDatum={anfangDatumQuery}
            endeDatum={endeDatumQuery}
          />
          {table === 'auswertung' && <AuswertungTabelle />}
        </Stack>
      </KalenderViewGrid>
    </>
  );
}

export type AntraegeRow = {
  einfuegungDatum: Dayjs;
  einfuegungTimestamp: IdTimestamp;
  typ: string;
  anfangTs: IdTimestamp;
  endeTs: IdTimestamp;
  anzahlTage?: number;
  zeitraum: string;
  status: AntragAbwesenheitStatus;
  antragSchluessel: string;
  antragGruppeId: string;
  mitarbeiterKommentar: string;
  fuehrungskraftKommentar: string;
  isGanztaegig: boolean;
  iconId: number;
  antragBezeichnung: string;
  absVgngPrzId: IdTimestamp;
  isStornierung: boolean;
};
export function parseReadAntraegeAbwesenheitToAntraegeTabelleRow(
  data: AntragAbwesenheitRead[],
  schluesselPropertiesMap: SchluesselPropertiesMap
): AntraegeRow[] {
  const { DATUM_FORMAT_STRING, UHRZEIT_FORMAT_STRING } = KalenderConstants;
  return data
    .map(
      ({
        einfuegungDatum,
        antragsSchluessel,
        timestampVon,
        timestampBis,
        isGanztaegig,
        anzahlTage,
        antragStatus,
        mitarbeiterKommentar,
        fuehrungskraftKommentar,
        absVgngPrzId,
        isStorniert,
      }) => {
        const schluesselProperties = schluesselPropertiesMap[antragsSchluessel];
        const dayjsBeginn = idTimestampToDayjs(timestampVon);
        const dayjsEnde = idTimestampToDayjs(timestampBis);
        if (schluesselProperties !== undefined) {
          return {
            einfuegungDatum: idTimestampToDayjs(einfuegungDatum),
            einfuegungTimestamp: einfuegungDatum,
            typ: isStorniert
              ? schluesselProperties.antragBezeichnung + ' - ' + t('Stornierung')
              : schluesselProperties.antragBezeichnung,
            anfangTs: timestampVon,
            endeTs: timestampBis,
            zeitraum: isGanztaegig
              ? `${dayjsBeginn.format(DATUM_FORMAT_STRING)} - ${dayjsEnde.format(DATUM_FORMAT_STRING)}`
              : `${dayjsBeginn.format(DATUM_FORMAT_STRING)} ${dayjsBeginn.format(UHRZEIT_FORMAT_STRING)} - ${dayjsEnde.format(UHRZEIT_FORMAT_STRING)}`,
            status: antragStatus,
            antragSchluessel: antragsSchluessel,
            antragGruppeId: schluesselProperties.gruppeId,
            iconId: schluesselProperties.iconId,
            antragBezeichnung: schluesselProperties.antragBezeichnung,
            anzahlTage,
            mitarbeiterKommentar,
            fuehrungskraftKommentar,
            isGanztaegig,
            absVgngPrzId,
            isStornierung: isStorniert,
          };
        }
        return undefined;
      }
    )
    .filter((item) => item !== undefined) as AntraegeRow[];
}

type SchluesselProperties = {
  iconId: number;
  kategorie: string;
  gruppe: string;
  gruppeId: string;
  antragBezeichnung: string;
};
type SchluesselPropertiesMap = Record<string, SchluesselProperties | undefined>;
export function getSchluesselToPropertiesMap(konfAntraege: KonfAntragskategorieRead[]): SchluesselPropertiesMap {
  const mapper: SchluesselPropertiesMap = {};
  for (const kategorie of konfAntraege) {
    for (const gruppe of kategorie.antragsgruppen) {
      for (const schluessel of gruppe.antraege) {
        mapper[schluessel.schluessel] = {
          kategorie: kategorie.bezeichnung,
          gruppe: gruppe.bezeichnung,
          gruppeId: gruppe.id,
          iconId: gruppe.iconId,
          antragBezeichnung: schluessel.bezeichnung,
        };
      }
    }
  }
  return mapper;
}

export default function AntraegeAbwesenheitTabelle({ hidden, anfangDatum, endeDatum }: ZeitereignisseTabelleProps) {
  const konfAntraege = useReadKonfAntraege().data?.antragskategorien ?? [];
  const defaultData = useReadAntraegeAbwesenheit(anfangDatum, endeDatum).data ?? [];

  const schluesselPropertiesMap = useMemo(() => getSchluesselToPropertiesMap(konfAntraege), [konfAntraege]);
  const parsedData = parseReadAntraegeAbwesenheitToAntraegeTabelleRow(defaultData, schluesselPropertiesMap);

  const workflowService = useWorkflowService();

  if (hidden) {
    return;
  }
  return (
    <AntraegeFITabelle
      rows={parsedData}
      onButtonClick={(antrag: AntraegeRow) => {
        navigateToAntragAbwesenheitAnzeigen(workflowService, {
          beginn: antrag.anfangTs,
          ende: antrag.endeTs,
          erstellungszeitpunkt: antrag.einfuegungTimestamp,
        });
      }}
    />
  );
}

type AntragStatusProps = {
  antragStatus?: AntragAbwesenheitStatus;
};

export function AntragStatus({ antragStatus }: AntragStatusProps) {
  switch (antragStatus) {
    case AntragAbwesenheitStatusConstants.STORNIERT:
      return <Chip icon={<IcCancelCircleS />} color="error" label={t('Storniert')} />;
    case AntragAbwesenheitStatusConstants.ABGELEHNT:
      return <Chip icon={<IcCancelCircleS />} color="error" label={t('abgelehnt')} />;

    case AntragAbwesenheitStatusConstants.FREIGEGEBEN:
    case AntragAbwesenheitStatusConstants.FREIGEGEBEN_SAP:
      return <Chip icon={<IcCheckCircleS />} color="success" label={t('genehmigt')} />;

    case AntragAbwesenheitStatusConstants.IN_GENEHMIGUNG:
    default:
      return <Chip icon={<IcCircleEmptyDotsS />} color="attention" label={t('inGenehmigung')} />;
  }
}

function AntraegeFITabelle({
  rows,
  onButtonClick,
}: {
  rows: AntraegeRow[];
  onButtonClick: (antrag: AntraegeRow) => void;
}) {
  const coldefs: FITabelleProps['colDefs'] = [
    {
      comparator: (valueA: Dayjs, valueB: Dayjs) => {
        if (valueA.isBefore(valueB)) {
          return -1;
        }
        if (valueA.isAfter(valueB)) {
          return 1;
        }
        return 0;
      },
      sort: 'desc',
      autoHeight: true,
      headerName: t('datum'),
      field: 'einfuegungDatum',
      cellRenderer: (params: ValueGetterParams<AntraegeRow>) =>
        params.data && (
          <Typography width="100%">
            {params.data.einfuegungDatum.format(KalenderConstants.DATUM_FORMAT_STRING)}
          </Typography>
        ),
    },
    {
      autoHeight: true,
      headerName: t('typ'),
      field: 'typ',
      cellStyle: () => ({ height: 'auto', justifyContent: 'center' }),
      cellRenderer: (params: ValueGetterParams<AntraegeRow>) =>
        params.data && (
          <Stack gap={2} alignItems="center" direction="row">
            <AntragIcon iconId={params.data.iconId} size={IconConstants.ICON_36} />
            {params.data.typ}
          </Stack>
        ),
    },
    {
      autoHeight: true,
      headerName: t('zeitenZeitraum'),
      field: 'zeitraum',
    },
    {
      autoHeight: true,
      headerName: t('statusZeitSaldo'),
      field: 'status',
      wrapText: true,
      cellRenderer: (params: ValueGetterParams<AntraegeRow>) =>
        params.data && (
          <Stack direction={'row'} justifyContent={'space-between'} alignItems={'center'}>
            <AntragStatus antragStatus={params.data.status} />
            <IconButton type="button" onClick={() => params.data && onButtonClick(params.data)} aria-label="next">
              <IcNextS />
            </IconButton>
          </Stack>
        ),
    },
  ];
  return (
    <FITabelle
      selektionsVariante="NONE"
      colDefs={coldefs}
      rows={rows}
      sideBar={false}
      getRowId={(params: GetRowIdParams<AntraegeRow>) => `${params.data.einfuegungDatum} ${params.data.zeitraum}`}
      disableSettingsSection={true}
      statusBar={undefined}
      minHeight={ThemeConstants.TABLE_MIN_HEIGHT}
    ></FITabelle>
  );
}
