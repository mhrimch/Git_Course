@AplComponent
public class AntragService implements IAntragService {

    private final transient UserService userService;

    private final transient PmgtPzwAntragTageErmRepository antragTageErmRepository;
    private final transient PmgtPzwAntragLesenRepository antragLesenRepository;
    private final transient PmgtPzwAntragStatusSetzRepository antragStatusSetzRepository;

    public AntragService(
            final UserService userService,
            final PmgtPzwAntragAnlegenRepository antragAnlegenRepository,
            final PmgtPzwAntragTageErmRepository antragTageErmRepository,
            final PmgtPzwAntragLesenRepository antragLesenRepository,
            final PmgtPzwAntragStatusSetzRepository antragStatusSetzRepository
    ) {
        this.userService = userService;
        this.antragAnlegenRepository = antragAnlegenRepository;
        this.antragTageErmRepository = antragTageErmRepository;
        this.antragLesenRepository = antragLesenRepository;
        this.antragStatusSetzRepository = antragStatusSetzRepository;
    }

    /**
     * Die Funktion createAntrag legt einen Antrag an.
     *
     * @throws UiServiceException UiServiceException
     */
    @AplLogging(logExecutionTime = true, includeParameters = true, logTrace = true)
    public void createAntrag(
            AntragAbwesenheitWrite antragAbwesenheitWrite
    ) throws UiServiceException {
        String personalnummer = userService.getPersonalnummer();
        LegPmgtPzwAntragAnUeberAbwesenheitEingabe eingabe = AntraegeMapper.antragWriteToOspe(antragAbwesenheitWrite, personalnummer);
        try {
            antragAnlegenRepository.legPmgtPzwAntragAnUeberAbwesenheit(eingabe);
        } catch (OspeException e) {
            throw new UiServiceException(MessageType.ERROR, MessageCode.ERROR,
                    "Antrag.Write.HttpStatus.502",
                    MessageMetadata.instance(MessageCallType.CREATE, MessageScope.APL));
        }
    }

    /**
     * Die Funktion antragAbwesenheitStatusSetzen ändert einen Antrag.
     *
     * @throws UiServiceException UiServiceException
     */
    @AplLogging(logExecutionTime = true, includeParameters = true, logTrace = true)
    public void updateAntragAbwesenheitStatus(
            AntragAbwesenheitStatusUpdate antragAbwesenheitStatusUpdate
    ) throws UiServiceException {
        String personalnummer = userService.getPersonalnummer();
        SetzPmgtPzwAntragStatusEingabe eingabe = AntraegeMapper.antragAbwesenheitStatusUpdateToOSPE(antragAbwesenheitStatusUpdate, personalnummer);
        try {
            antragStatusSetzRepository.setzPmgtPzwAntragStatus(eingabe);
        } catch (OspeException e) {
            throw new UiServiceException(MessageType.ERROR, MessageCode.ERROR,
                    "Antrag.Update.HttpStatus.502",
                    MessageMetadata.instance(MessageCallType.CREATE, MessageScope.APL));
        }
    }

    /**
     * Die Funktion validateUrlaubszeitraum validiert Tage und Resturlaubstage für einen Urlaubsantrag.
     *
     * @return UrlaubszeitraumValidateResponse
     */
    @AplLogging(logExecutionTime = true, includeParameters = true, logTrace = true)
    public AntragValidateRead validateUrlaubszeitraum(
            IdTimestamp anfangTimestamp,
            IdTimestamp endeTimestamp
    ) throws UiServiceException {
        ErmittlePmgtPzwAntragTageUeberPctlPersSchlEingabe eingabe = AntraegeMapper.createAntragTageUeberPctlPersSchlEingabe(anfangTimestamp, endeTimestamp, userService.getPersonalnummer());
        try {
            return antragTageErmRepository.ermittlePmgtPzwAntragTageUeberPctlPersSchl(eingabe)
                    .map(AntraegeMapper::ospeToAntragValidate)
                    .orElse(null);
        } catch (OspeException e) {
            throw new UiServiceException(MessageType.ERROR, MessageCode.ERROR,
                    "Antrag.Validation.HttpStatus.502",
                    MessageMetadata.instance(MessageCallType.CREATE, MessageScope.APL));
        }
    }

    /**
     * Die Funktion liest die Anträge des aufrufenden Users.
     *
     * @return List<AntragAbwesenheitRead>
     */
    @AplLogging(logExecutionTime = true, includeParameters = true, logTrace = true)
    public List<AntragAbwesenheitRead> readAntraegeAbwesenheit(
            LocalDate beginn,
            LocalDate ende
    ) throws UiServiceException {
        try {
            LocalDateTime localDateTimeAnfang = beginn.atTime(0, 0, 0);
            LocalDateTime localDateTimeEnde = ende.atTime(23, 59, 59);
            DateTimeFormatter tsFormatter = DateTimeFormatter.ofPattern((DateFormatConstants.TIMESTAMP_FORMAT), Locale.GERMANY);
            LiesPmgtPzwAntragUeberUserEingabe eingabe = new LiesPmgtPzwAntragUeberUserEingabe();
            eingabe.setPctlPersSchl(userService.getPersonalnummer());
            eingabe.setTsGueltigAb(new IdTimestamp(localDateTimeAnfang.format(tsFormatter)));
            eingabe.setTsGueltigBis(new IdTimestamp(localDateTimeEnde.format(tsFormatter)));
            eingabe.setPzwAntrArt(PzwAntraegeConstants.ANTRAGSART_ABWESENHEITEN);
            return antragLesenRepository
                    .liesPmgtPzwAntragUeberUser(eingabe)
                    .map(AntraegeMapper::ospeToantragAbwesenheitRead)
                    .orElse(null);
        } catch (OspeException e) {
            throw new UiServiceException(MessageType.ERROR, MessageCode.ERROR,
                    "Antraege.Read.HttpStatus.502",
                    MessageMetadata.instance(MessageCallType.READ, MessageScope.APL));
        }
    }
}

package osplus.pzw.mapper;

import osplus.dynssee.zeitwirtschaft.fo.ErmittlePmgtPzwAntragTageUeberPctlPersSchlEingabe;
import osplus.dynssee.zeitwirtschaft.fo.LegPmgtPzwAntragAnUeberAbwesenheitEingabe;
import osplus.dynssee.zeitwirtschaft.fo.PmgtPzwAntragLesenAusgabe;
import osplus.dynssee.zeitwirtschaft.fo.PmgtPzwAntragTageErmAusgabe;
import osplus.dynssee.zeitwirtschaft.fo.SetzPmgtPzwAntragStatusEingabe;
import osplus.josp.types.IdTimestamp;
import osplus.psk.PskConstants;
import osplus.pzw.model.antrag.Abwesenheitszyklus;
import osplus.pzw.model.antrag.AntragAbwesenheitRead;
import osplus.pzw.model.antrag.AntragAbwesenheitStatus;
import osplus.pzw.model.antrag.AntragAbwesenheitStatusUpdate;
import osplus.pzw.model.antrag.AntragAbwesenheitWrite;
import osplus.pzw.model.antrag.AntragValidateRead;

import java.util.List;
import java.util.Optional;

public class AntraegeMapper {
    public static LegPmgtPzwAntragAnUeberAbwesenheitEingabe antragWriteToOspe(AntragAbwesenheitWrite antragAbwesenheitWrite, String personalNummer) {
        LegPmgtPzwAntragAnUeberAbwesenheitEingabe eingabe = new LegPmgtPzwAntragAnUeberAbwesenheitEingabe();
        eingabe.setPctlPersSchl(personalNummer);
        eingabe.setPzwAbwtSchl(antragAbwesenheitWrite.abwesenheitsgrundId());
        eingabe.setPzwAntrStrnKz(antragAbwesenheitWrite.stornierungskennzeichen());
        eingabe.setPzwAntrZrmAts(antragAbwesenheitWrite.anfangTimestamp());
        eingabe.setPzwAntrZrmEts(antragAbwesenheitWrite.endeTimestamp());
        eingabe.setPzwAntrZyklSchl(antragAbwesenheitWrite.abwesenheitszyklus().getZyklusSchluessel());
        eingabe.setPzwAntrMaHnws(antragAbwesenheitWrite.mitarbeiterHinweis());
        eingabe.setMassVgngId(antragAbwesenheitWrite.dokumentId());
        return eingabe;
    }

    public static SetzPmgtPzwAntragStatusEingabe antragAbwesenheitStatusUpdateToOSPE(AntragAbwesenheitStatusUpdate request, String nutzerPersonalnummer) {
        SetzPmgtPzwAntragStatusEingabe eingabe = new SetzPmgtPzwAntragStatusEingabe();
        eingabe.setPctlPersSchl(nutzerPersonalnummer);
        eingabe.setPzwAntrArt(request.antragArt());
        eingabe.setTsEinfuegung(request.antragEinfuegungTimestamp());
        eingabe.setPzwAntrStat(request.antragStatus());
        eingabe.setAbsVgngAtgId(request.absVgngAtgId());
        eingabe.setPzwAntrFrgbHnws(request.freigabeHinweis());
        eingabe.setAbsAtgName(request.absAtgName());
        Optional.ofNullable(request.mitarbeiterPersonalnummer()).ifPresent(eingabe::setMaPctlPersSchl);
        if (request.nutzerIstFuehrungskraft()) {
            eingabe.setFrkrPctlPersSchl(nutzerPersonalnummer);
        }
        return eingabe;
    }

    public static List<AntragAbwesenheitRead> ospeToantragAbwesenheitRead(PmgtPzwAntragLesenAusgabe ausgabe) {
        return Optional.ofNullable(ausgabe.getPmgtPzwAntragMitarbeiter())
                .stream()
                .flatMap(antragMitarbeiter -> antragMitarbeiter.getPmgtPzwAntragAbwht().stream())
                .map(antrag -> new AntragAbwesenheitRead(
                        antrag.getPzwAntrZrmAts(),
                        antrag.getPzwAntrZrmEts(),
                        PskConstants.JA.equals(antrag.getPzwAntrStrnKz()),
                        antrag.getPzwAbwtSchl(),
                        antrag.getPzwAntrTageAnz().toBigInteger().intValue(),
                        Abwesenheitszyklus.GANZTAGS.getZyklusSchluessel().equals(antrag.getPzwAntrZyklSchl()),
                        antrag.getPzwAntrMaHnws(),
                        antrag.getPzwAntrFrgbHnws(),
                        antrag.getTsEinfuegung(),
                        AntragAbwesenheitStatus.getAntragStatus(antrag.getPzwAntrStat()),
                        antrag.getAbsVgngPrzId()
                )).toList();
    }

    public static AntragValidateRead ospeToAntragValidate(PmgtPzwAntragTageErmAusgabe ausgabe) {
        return new AntragValidateRead(
                ausgabe.getPzwAntrTageAnz().intValue(),
                ausgabe.getPzwUrlbTageRest().intValue()
        );
    }

    public static ErmittlePmgtPzwAntragTageUeberPctlPersSchlEingabe createAntragTageUeberPctlPersSchlEingabe(
            IdTimestamp anfangTimestamp,
            IdTimestamp endeTimestamp,
            String personalNummer
    ) {
        ErmittlePmgtPzwAntragTageUeberPctlPersSchlEingabe eingabe = new ErmittlePmgtPzwAntragTageUeberPctlPersSchlEingabe();
        eingabe.setPzwAntrZrmAts(anfangTimestamp);
        eingabe.setPzwAntrZrmEts(endeTimestamp);
        eingabe.setPctlPersSchl(personalNummer);
        return eingabe;
    }
}
