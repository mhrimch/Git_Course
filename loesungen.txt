type ZeitkontoRow = {
  datum: Dayjs;
  typ: string;
  zeitraum?: string;
  anfangTs?: string;
  endeTs?: string;
  iconId?: number;
  status: ZeitkontoStatusProps['label'];
};

export type ZeitkontoTabelleProps = {
  items: ZeitkontoRow[];
};

function mapKalenderReadToZeitkontoRow(kalendertage: KalendertagRead[]) {
  const rows: ZeitkontoRow[] = [];
  for (const tag of kalendertage) {
    const { datum, buchungspaare, tagesSaldo } = tag;
    const dayjsDatum = dayjs(datum);
    for (const abwesenheit of tag.antraege) {
      if (abwesenheit.beginn && abwesenheit.ende && abwesenheit.art !== AnOderAbwesenheitsarten.anwesend) {
        rows.push({
          datum: dayjsDatum,
          zeitraum: `${convertISOTimestampToDate(abwesenheit.beginn)} - ${convertISOTimestampToDate(abwesenheit.ende)}`,
          anfangTs: dayjsToIdTimestamp(dayjs(abwesenheit.beginn)),
          endeTs: dayjsToIdTimestamp(dayjs(abwesenheit.ende)),
          iconId: abwesenheit.iconNr,
          status: 'Ganztägig',
          typ: abwesenheit.art,
        });
      }
    }
    if (tag.buchungspaare.length > 0) {
      const convertedSaldo = convertDecimalToHHmm(tagesSaldo.toString());
      let status: ZeitkontoStatusProps['label'] =
        tagesSaldo < 0 ? (convertedSaldo as `-${string}`) : `+${convertedSaldo}`;
      if (tagesSaldo === 0) {
        status = 'Ganztägig';
      }
      rows.push({
        datum: dayjsDatum,
        zeitraum: `${buchungspaare[0].beginn} - ${buchungspaare[buchungspaare.length - 1].ende}`,
        typ: AnOderAbwesenheitsarten.anwesend,
        status: status,
      });
    }
  }
  return rows;
}

type ZeitkontoStatusProps = {
  label: 'Ganztägig' | `+${string}` | `-${string}` | `aktiv seit${string}`;
};

function ZeitkontoStatus({ label }: ZeitkontoStatusProps) {
  if (label.startsWith(ZeitKontoStatusConstants.MINUS)) {
    return <Chip icon={<IcClockS />} label={label} color="error" />;
  }
  if (label.startsWith(ZeitKontoStatusConstants.PLUS)) {
    return <Chip icon={<IcClockS />} label={label} color="success" />;
  }
  if (label.toLocaleLowerCase() === ZeitKontoStatusConstants.GANZTAEGIG) {
    return <Chip label={label} variant="outlined" color="secondary" />;
  }
  return <Chip icon={<IcClockS />} label={label} />;
}

  const start = dayjs(abwesenheit.beginn);
  const end = dayjs(abwesenheit.ende);
  const startDate = start.format("DD.MM.YY");
  const endDate = end.format("DD.MM.YY");
  const isSameDay = start.isSame(end, "day");
  // Untertägig = gleicher Tag, aber nicht 00:00 bis 23:xx
  const isUntertaegig =
    isSameDay &&
    (start.hour() !== 0 || end.hour() !== 23);
  // Zeitraum bauen
  let zeitraum;
  if (!isSameDay) {
    // Mehrtägig → immer ganztägig
    zeitraum = `${startDate} - ${endDate}`;
  } else if (isUntertaegig) {
    zeitraum = `${startDate} ${start.format("HH:mm")} - ${end.format("HH:mm")}`;
  } else {
    zeitraum = `${startDate}`;
  }
  // Status bestimmen
  let status: ZeitkontoStatusProps['label'];
  if (isUntertaegig) {
    // Dauer berechnen (HH:mm)
    const diffMinutes = end.diff(start, "minute");
    const hours = Math.floor(diffMinutes / 60)
      .toString()
      .padStart(2, "0");
    const minutes = (diffMinutes % 60).toString().padStart(2, "0");
    status = `+${hours}:${minutes}`;
  } else {
    // Ganztägig
    status = "Ganztägig";
  }
  rows.push({
    datum: start,
    zeitraum,
    anfangTs: dayjsToIdTimestamp(start),
    endeTs: dayjsToIdTimestamp(end),
    iconId: abwesenheit.iconNr,
    status,
    typ: abwesenheit.art,
  });

function ZeitkontoStatus({ label, typ }: ZeitkontoStatusProps) {
  const isAnwesenheit = typ === AnOderAbwesenheitsarten.anwesend;
  // 1. Negativer Wert -> rotes Icon
  if (label.startsWith("-")) {
    return <Chip icon={<IcClockS />} label={label} color="error" />;
  }
  // 2. Positiver Wert -> grünes Icon
  if (label.startsWith("+")) {
    return <Chip icon={<IcClockS />} label={label} color="success" />;
  }
  // 3. Ganztägig
  if (label.toLowerCase() === ZeitKontoStatusConstants.GANZTAEGIG) {
    if (isAnwesenheit) {
      // Anwesenheit -> GRÜNES ICON
      return <Chip icon={<IcClockS />} label={label} color="success" />;
    }
    // Abwesenheit -> OUTLINED, wie bisher
    return <Chip label={label} variant="outlined" color="secondary" />;
  }
  // 4. Anwesenheit ohne + oder - -> Icon (z. B. 0:19)
  if (isAnwesenheit) {
    return <Chip icon={<IcClockS />} label={label} color="success" />;
  }
  // 5. Untertägige Abwesenheit (Dauer wie "1:26") -> kein Icon
  return <Chip label={label} />;
}

function ZeitkontoStatus({ label: initialLabel, typ }: ZeitkontoStatusProps) {
  const isAnwesenheit = typ === AnOderAbwesenheitsarten.anwesend;
  let label = initialLabel;
  // 0. Bei Anwesenheit IMMER + hinzufügen, außer bei Ganztägig oder -Werten
  if (
    isAnwesenheit &&
    !label.startsWith("+") &&
    !label.startsWith("-") &&
    label.toLowerCase() !== ZeitKontoStatusConstants.GANZTAEGIG
  ) {
    label = `+${label}`;
  }
  // 1. Negative Werte -> rotes Icon
  if (label.startsWith("-")) {
    return <Chip icon={<IcClockS />} label={label} color="error" />;
  }
  // 2. Positive Werte -> grünes Icon
  if (label.startsWith("+")) {
    return <Chip icon={<IcClockS />} label={label} color="success" />;
  }
  // 3. Ganztägig
  if (label.toLowerCase() === ZeitKontoStatusConstants.GANZTAEGIG) {
    if (isAnwesenheit) {
      return <Chip icon={<IcClockS />} label={label} color="success" />;
    }
    return <Chip label={label} variant="outlined" color="secondary" />;
  }
  // 4. Untertägige Abwesenheit -> kein Icon
  return <Chip label={label} />;
}
